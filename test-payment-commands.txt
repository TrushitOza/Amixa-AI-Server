# Payment System Testing Commands

# First, add these to your .env file:
# RAZORPAY_KEY_ID=rzp_test_your_key_id_here
# RAZORPAY_KEY_SECRET=your_key_secret_here
# RAZORPAY_WEBHOOK_SECRET=your_webhook_secret_here

# For testing without real Razorpay credentials, you can use dummy values:
# RAZORPAY_KEY_ID=rzp_test_dummy_key_id
# RAZORPAY_KEY_SECRET=dummy_key_secret

# ===== AUTHENTICATION FIRST =====
# 1. Register a user
curl -X POST http://localhost:5000/api/v1/auth/register \
  -H "Content-Type: application/json" \
  -d '{
    "firstname": "Test",
    "lastname": "User",
    "email": "test@example.com",
    "password": "password123"
  }'

# 2. Login to get JWT token
curl -X POST http://localhost:5000/api/v1/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "email": "test@example.com",
    "password": "password123"
  }'

# Copy the token from login response and use it in Authorization header below
# Replace YOUR_JWT_TOKEN with actual token

# ===== PAYMENT SYSTEM TESTING =====

# 1. Get all pricing plans (Public - No auth required)
curl -X GET http://localhost:5000/api/v1/payments/plans

# 2. Get pricing plans by type
curl -X GET "http://localhost:5000/api/v1/payments/plans?planType=one_time"
curl -X GET "http://localhost:5000/api/v1/payments/plans?planType=monthly"
curl -X GET "http://localhost:5000/api/v1/payments/plans?planType=yearly"

# 3. Get payment configuration (Public)
curl -X GET http://localhost:5000/api/v1/payments/config

# 4. Create payment order (Requires auth)
curl -X POST http://localhost:5000/api/v1/payments/create-order \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN" \
  -d '{
    "planName": "starter_pack"
  }'

# 5. Get payment order details
curl -X GET http://localhost:5000/api/v1/payments/order/RAZORPAY_ORDER_ID \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"

# 6. Simulate payment verification (Use order_id from step 4)
curl -X POST http://localhost:5000/api/v1/payments/verify \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN" \
  -d '{
    "razorpay_order_id": "order_from_step_4",
    "razorpay_payment_id": "pay_test_payment_id",
    "razorpay_signature": "test_signature",
    "payment_method": "upi"
  }'

# 7. Handle payment failure
curl -X POST http://localhost:5000/api/v1/payments/failure \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN" \
  -d '{
    "razorpay_order_id": "order_from_step_4",
    "failure_reason": "Payment cancelled by user"
  }'

# 8. Get user payment history
curl -X GET http://localhost:5000/api/v1/payments/history \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"

# 9. Get payment history with filters
curl -X GET "http://localhost:5000/api/v1/payments/history?page=1&limit=10&status=paid&planType=one_time" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"

# 10. Create Razorpay customer
curl -X POST http://localhost:5000/api/v1/payments/create-customer \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"

# ===== CREDIT SYSTEM TESTING =====

# 11. Get user credit information
curl -X GET http://localhost:5000/api/v1/credits \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"

# 12. Get user transaction history
curl -X GET http://localhost:5000/api/v1/transactions \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"

# 1. Register/Login user
# 2. Seed pricing plans
# 3. Get pricing plans
# 4. Create payment order
# 5. Verify payment (this will add credits to user)
# 6. Check user credits
# 7. Generate image (this will consume credits)
# 8. Check transaction history

# ===== IMAGE GENERATION TESTING (After purchasing credits) =====

# Generate image (consumes 1 credit)
curl -X POST http://localhost:5000/api/v1/images/generate \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN" \
  -d '{
    "prompt": "A beautiful sunset over mountains",
    "style": "realistic"
  }'

# Generate logo (consumes 1 credit)
curl -X POST http://localhost:5000/api/v1/images/logo \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_JWT_TOKEN" \
  -d '{
    "prompt": "Modern tech company logo",
    "style": "minimalist"
  }'

# ===== WEBHOOK TESTING =====
# Simulate Razorpay webhook (for payment success)
curl -X POST http://localhost:5000/api/v1/payments/webhook \
  -H "Content-Type: application/json" \
  -H "X-Razorpay-Signature: test_signature" \
  -d '{
    "event": "payment.captured",
    "payload": {
      "payment": {
        "entity": {
          "id": "pay_test_payment_id",
          "order_id": "order_test_order_id",
          "method": "upi",
          "status": "captured"
        }
      }
    }
  }'
